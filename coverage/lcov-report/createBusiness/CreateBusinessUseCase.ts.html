import { inject, injectable } from 'tsyringe';

import { IBusiness } from '@entities/IBusiness';
import { Role_Permissao } from '@entities/IPermissao';
import { BadRequestError, NotAuthorizedError } from '@imobo/nano-common';
import { IBusinessRepository } from '@repositories/IBusinessRepository';
import { IMailProvider } from '@shared/container/providers/MailProvider/IMailProvider';
import {
  FileType,
  IStorageProvider,
} from '@shared/container/providers/StorageProvider/IStorageProvider';
import { AUTH_ACTIVATE_MAIL_SENDER } from '@utils/constants';
import { DEFAULT_BUSINESS_LOGO } from '@utils/default-logo';
import { getTemplateEmail } from '@utils/get-template-email';

import { IRequest } from './types';

@injectable()
class CreateBusinessUseCase {
  constructor(
    @inject('BusinessRepository')
    private businessRepository: IBusinessRepository,
    @inject('MailProvider')
    private mailProvider: IMailProvider,
    @inject('StorageProvider')
    private storageProvider: IStorageProvider
  ) {}

  async execute({
    file,
    cnpj,
    razao,
    fantasia,
    vinculo,
    emailcontato,
    telefonecontato,
    url,
    usuario,
    currentUser,
  }: IRequest): Promise<IBusiness> {
    const businessExiste = await this.businessRepository.findByCnpj(cnpj);

    if (businessExiste) {
      throw new BadRequestError('CNPJ informado já está em uso.');
    }

    const additionalData = {};

    if (vinculo) {
      if (!currentUser) {
        throw new BadRequestError(
          'Para criar um business vinculado é necessário estar logado'
        );
      }
      if (
        !currentUser.permissions.includes(Role_Permissao.AUTH_BUSINESS_MULT)
      ) {
        throw new NotAuthorizedError();
      }

      if (!currentUser.business?.id) {
        throw new BadRequestError(
          'Usuário precisa estar vinculado a um business.'
        );
      }

      const userBusiness = await this.businessRepository.findById(
        currentUser.business?.id
      );

      if (!userBusiness) {
        throw new BadRequestError('Não foi possível vincular o business.');
      }

      Object.assign(additionalData, {
        cdBusinessAdm: userBusiness.cdBusiness,
        cdWorkflow: userBusiness.cdWorkflow,
      });
    }

    const business = await this.businessRepository.create({
      nmFantasia: fantasia,
      nmRazao: razao,
      nuCnpj: cnpj,
      url,
      deEmail: emailcontato,
      nuTelefone: telefonecontato,
      cdWorkflow: 'default',
      usuarios: [usuario],
      ...additionalData,
    });

    await this.mailProvider.sendMail({
      sender: AUTH_ACTIVATE_MAIL_SENDER,
      recipient: usuario.deEmail,
      subject: 'Ative sua conta no Nano.',
      body: await getTemplateEmail('ativacao', String(usuario.deToken)),
    });

    const path = await this.storageProvider.uploadFile({
      file: file || DEFAULT_BUSINESS_LOGO,
      id: business.cdBusiness,
      type: FileType.BUSINESS,
    });

    await this.businessRepository.changeImage(business, path);

    return business;
  }
}

export { CreateBusinessUseCase };
